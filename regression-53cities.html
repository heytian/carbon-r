Regression analysis (53 cities)

Carbon impact score (CO2 conc - CO2 emissions)

  • A ‘carbon impact score’ is introduced by taking the normalized interquartile range of CO2 concentration levels (OCO-3 netcdf data) subtracted by CO2 emissions levels (satellite-based estimates from Ahn et al, 2025).

  • Higher score = more “harmed” (high conc low emissions), lower score = more “harming” done (low conc high emissions)

  • This is then compared against solar induced fluorescence (SIF), population, gdp, number of samples, and coastal proximity via running a multivariate linear regression.

# install.packages("tidyverse")
library(tidyverse)


df <- read.csv("c40cities-co2-em-conc-wkt-coast.csv") # outcome of https://github.com/heytian/d2d-oco3-tools/blob/main/20260405_04_wkt-coastal.py
df <- df |> drop_na(population)

# normalize independent variables using interquartile range 
iqr_norm <- function(x) {
  q1 <- quantile(x, 0.25, na.rm = TRUE)
  q3 <- quantile(x, 0.75, na.rm = TRUE)
  (x - q1) / (q3 - q1)
}


norm_df <- df |> 
  mutate(
    co2_conc_iqr = iqr_norm(xco2_ppm),
    co2_em_iqr = iqr_norm(annual_co2_em_mtco2),
    sif_iqr = iqr_norm(sif_757nm),
    pop_iqr = iqr_norm(population),
    xco2numsams_iqr = iqr_norm(xco2_numsams),
    sifnumsams_iqr = iqr_norm(sif_numsams),
    gdp_iqr = iqr_norm(gdp_bil_usd),
    coast_iqr = iqr_norm(coastal_km),
    impact_iqr = co2_conc_iqr - co2_em_iqr
  )
    

lm_iqr_impact<- lm(impact_iqr ~ sif_iqr + pop_iqr + xco2numsams_iqr + sifnumsams_iqr + gdp_iqr + coast_iqr, 
                  data = norm_df)
summary(lm_iqr_impact)

# bubble size = population, color = quadrant, shape = coastal/inland
ggplot(norm_df, aes(x = co2_em_iqr, y = co2_conc_iqr)) +
  
  # quadrant dividing lines at IQR thresholds (0 = Q1, 1 = Q3)
  geom_hline(yintercept = 0, linetype = "dashed", color = "gray60") +
  geom_hline(yintercept = 1, linetype = "dashed", color = "gray60") +
  geom_vline(xintercept = 0, linetype = "dashed", color = "gray60") +
  geom_vline(xintercept = 1, linetype = "dashed", color = "gray60") +
  
  # cities as bubbles
  geom_point(aes(
    color = quadrant,
    size  = population / 1e6, # size by population millions
    shape = is_coastal
  ), alpha = 0.75) +
  
  # label interesting cities only
  ggrepel::geom_text_repel(
    data = norm_df |> filter(
      quadrant %in% c("LOW_emissions_HIGH_conc", "HIGH_emissions_LOW_conc")
    ),
    aes(label = city),
    size = 2.8, max.overlaps = 15
  ) +
  
  scale_color_manual(values = c(
    "HIGH_both"              = "#d73027",
    "LOW_both"               = "#4575b4",
    "HIGH_emissions_LOW_conc" = "#fc8d59",
    "LOW_emissions_HIGH_conc" = "#91bfdb")) +
  scale_shape_manual(values = c("True" = 17, "False" = 16),
                     labels = c("True" = "Coastal", "False" = "Inland")) +
  scale_size_continuous(range = c(2, 10), name = "Population (M)") +
  
  labs(
    title    = "City CO2 Concentration vs Emissions",
    subtitle = "Quadrants based on IQR thresholds | Size = population | Shape = coastal/inland",
    x        = "CO2 Emissions (IQR normalized)",
    y        = "CO2 Concentration (IQR normalized)",
    color    = "Quadrant",
    shape    = NULL
  ) +
  theme_minimal() +
  theme(legend.position = "right")

filtered_data <- oco3_full |>
  filter(
    season %in% c("Tropical Wet", "Summer (North)"),
    time_of_day %in% c("Morning", "Afternoon"),

iqr_norm <- function(x) {
  q1 <- quantile(x, 0.25, na.rm = TRUE)
  q3 <- quantile(x, 0.75, na.rm = TRUE)
  (x - q1) / (q3 - q1)

regression_df <- filtered_data |>
  group_by(target_name, country, population, gdp_bil_usd, coastal_km, season, time_of_day) |>
  summarise(
    country = first(country),
    population = first(population),
    gdp_bil_usd = first(gdp_bil_usd),
    coastal_km = first(coastal_km),
    xco2_median = median(xco2, na.rm = TRUE),
    xco2_std = sd(xco2, na.rm = TRUE),
    xco2_n = n(),
    sif_median = median(Daily_SIF_757nm, na.rm = TRUE),
    sif_std = sd(Daily_SIF_757nm, na.rm = TRUE),
    sif_n = n(),
    .groups = "drop"
  ) |>
  filter(!is.na(population)) |>
  mutate(
    xco2_iqr = iqr_norm(xco2_median),
    sif_iqr = iqr_norm(sif_median),
    pop_iqr = iqr_norm(population),
    xco2_numsams_iqr = iqr_norm(xco2_n),
    sif_numsams_iqr = iqr_norm(sif_n),
    gdp_iqr = iqr_norm(gdp_bil_usd),
    coast_iqr = iqr_norm(coastal_km),
    season = as.factor(season),
    time_of_day = as.factor(time_of_day)
  )
